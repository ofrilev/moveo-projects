
Project definition:

create table
  public.projects (
    id integer generated by default as identity not null,
    created_at timestamp with time zone not null default now(),
    title character varying not null,
    description character varying null,
    userid character varying not null,
    role smallint not null default '2'::smallint,
    constraint projects_pkey primary key (id),
    constraint projects_id_key unique (id)
  ) tablespace pg_default;

  Tasks definition:

  
  create table
  public.tasks (
    id integer generated by default as identity not null,
    created_at timestamp with time zone not null default now(),
    projectid integer not null,
    userid character varying not null,
    status smallint not null default '0'::smallint,
    title character varying not null,
    description character varying null,
    role smallint not null default '2'::smallint,
    constraint tasks_pkey primary key (id),
    constraint tasks_id_key unique (id),
    constraint tasks_projectid_fkey foreign key (projectid) references projects (id) on delete cascade
  ) tablespace pg_default;
